# 预览点击定位编辑实现方案（方案A）

## 目标

实现点击预览区元素，自动定位到编辑器对应位置。

## 实现步骤

### 步骤 1：修改 Markdown 渲染，注入位置信息

**文件**: `MarkdownViewer.tsx` 或渲染 Markdown 的组件

```typescript
// 1.1 修改渲染函数，为每个块添加 data-source-line 属性
function renderWithSourceMap(content: string): string {
  // 使用 marked 的 lexer 获取 token 和位置信息
  const tokens = marked.lexer(content)
  let currentLine = 1
  
  const html = tokens.map(token => {
    const line = currentLine
    const lineCount = token.raw.split('\n').length
    currentLine += lineCount
    
    // 根据 token 类型渲染，注入行号信息
    return renderToken(token, line)
  }).join('')
  
  return html
}

// 1.2 为每种 token 类型添加 data 属性
function renderToken(token: Token, startLine: number): string {
  const attrs = `data-source-line="${startLine}" data-source-type="${token.type}" class="md-block"`
  
  switch (token.type) {
    case 'heading':
      return `<h${token.depth} ${attrs}>${token.text}</h${token.depth}>`
    case 'paragraph':
      return `<p ${attrs}>${token.text}</p>`
    case 'list':
      return `<ul ${attrs}>${token.items.map(i => `<li>${i.text}</li>`).join('')}</ul>`
    case 'code':
      return `<pre ${attrs}><code>${token.text}</code></pre>`
    // ... 其他类型
    default:
      return token.raw
  }
}
```

### 步骤 2：预览区添加点击事件

**文件**: `MarkdownViewer.tsx`

```typescript
// 2.1 在预览容器上绑定点击事件
function Preview({ content, onNavigateToLine }: PreviewProps) {
  const containerRef = useRef<HTMLDivElement>(null)
  
  // 渲染带位置信息的 HTML
  const html = useMemo(() => renderWithSourceMap(content), [content])
  
  // 2.2 点击处理
  const handleClick = useCallback((e: MouseEvent) => {
    // 向上查找最近的 md-block 元素
    const block = (e.target as HTMLElement).closest('.md-block')
    if (!block) return
    
    const line = parseInt(block.getAttribute('data-source-line') || '1')
    onNavigateToLine(line)
  }, [onNavigateToLine])
  
  useEffect(() => {
    const container = containerRef.current
    if (!container) return
    
    container.addEventListener('click', handleClick)
    return () => container.removeEventListener('click', handleClick)
  }, [handleClick])
  
  return (
    <div 
      ref={containerRef}
      className="markdown-preview"
      dangerouslySetInnerHTML={{ __html: html }}
    />
  )
}
```

### 步骤 3：编辑器暴露跳转接口

**文件**: 编辑器组件（如 `Editor.tsx`）

```typescript
// 3.1 添加跳转到指定行的方法
interface EditorRef {
  navigateToLine: (line: number) => void
}

const Editor = forwardRef<EditorRef, EditorProps>((props, ref) => {
  const editorInstance = useRef<monaco.editor.IStandaloneCodeEditor>()
  
  useImperativeHandle(ref, () => ({
    navigateToLine: (line: number) => {
      if (!editorInstance.current) return
      
      // 滚动到指定行并选中
      editorInstance.current.revealLineInCenter(line)
      editorInstance.current.setPosition({ lineNumber: line, column: 1 })
      editorInstance.current.focus()
    }
  }))
  
  // ... 编辑器初始化
})
```

### 步骤 4：父子组件连接

**文件**: `WorkspaceShell.tsx` 或包含编辑器和预览的父组件

```typescript
function WorkspaceShell() {
  const editorRef = useRef<EditorRef>(null)
  
  // 4.1 处理预览点击，通知编辑器跳转
  const handleNavigateToLine = useCallback((line: number) => {
    editorRef.current?.navigateToLine(line)
  }, [])
  
  return (
    <div className="workspace">
      <Editor ref={editorRef} content={content} onChange={setContent} />
      <Preview content={content} onNavigateToLine={handleNavigateToLine} />
    </div>
  )
}
```
